//program to find the circumference of a circle
var readline = require('readline');
var input = readline.createInterface({
    input: process.stdin
});
var userInput=[];
input.on('line', (values) => {
    userInput.push(values);
});
input.on('close', () => {
    var a = userInput[0];
    var result=0;
    if(a<0)
    console.log("Error");
    else{
        result=(2*Math.PI*a).toFixed(2);
        console.log(result);
    }
});


Input:
======
2

Output:
=======
12.57
