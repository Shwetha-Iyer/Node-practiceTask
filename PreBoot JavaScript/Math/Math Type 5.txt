// program to find the factorial of a number
var readline = require('readline');
var input = readline.createInterface({
    input: process.stdin
});
var userInput=[];
input.on('line', (values) => {
    userInput.push(values);
});
input.on('close', () => {
    var n = parseInt(userInput[0]);
    var fact=1;
    for(var i=1;i<=n;i++){
        fact*=i;
    }
    console.log(fact);
});

Input:
======
5

Output:
=======
120
