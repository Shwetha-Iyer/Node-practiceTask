// program to print the sum of digits of a number
var readline = require('readline');
var input = readline.createInterface({
    input: process.stdin
});
var userInput=[];
input.on('line', (values) => {
    userInput.push(values);
});
input.on('close', () => {
    var n = parseInt(userInput[0]);
    var sum=0;
    var temp=0;
    while(n>0){
        temp=n%10;
        sum+=temp
        n=parseInt(n/10);
    }
    console.log(sum);
});


Input:
======
67654342345

Output:
=======
49
