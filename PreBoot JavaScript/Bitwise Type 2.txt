// program find the bitwise not of a number
var readline = require('readline');
var input = readline.createInterface({
    input : process.stdin
});
var userInput = [];
input.on('line', (values) => {
    userInput.push(values);
});
input.on('close', () => {
    var n = userInput[0];
    console.log(~n);
});

Input:
=====
5

Output:
======
-6
